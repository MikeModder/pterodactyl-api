const crypto = require('crypto');
const request = require('request-promise');

class APIWrapper {
    /**
     * A new API interface
     * @constructor
     * @param {String} apiurl - Base API URL
     * @param {String} pubkey - Public key for auth
     * @param {String} privkey - Private key for auth
     */
    constructor(apiurl, pubkey, privkey) {
        if(!apiurl || !pubkey || !privkey) throw new Error('You must pass a url. pubkey, and privkey to the constructor!');
        this.url = apiurl;
        this.pubkey = pubkey;
        this.privkey = privkey;
    }

    /**
     * Generates the authentication header.
     * @async
     * @param {String} url - URL
     * @param {String} body - Body
     * @param {String} pubkey - Public key
     * @param {String} privkey - Private key
     * @returns {String} Built auth header
     * @throws {Error} Any errors
     */
    getAuthHeader(url, body, pubkey, privkey) {
        return new Promise(async (resolve, reject) => {
            if(!url || !body || !pubkey || !privkey) reject(new Error('Missing one or more parameters!'));
            let hmac = crypto.createHmac('sha256', privkey);
            hmac.write(url + body);
            hmac.end();
            let hash = hmac.read().toString('base64');
            resolve(`${pubkey}.${hash}`);
        });
    }

    // User related functions

    /**
     * Get all users
     * @async
     * @returns {PTUser[]} - Array of users
     * @throws {Error} - Any errors that happen
     */
    getUsers() {
        return new Promise(async (resolve, reject) => {
            const url = this.url + '/api/admin/users';
            try {
                const auth = await this.getAuthHeader(url, '', this.pubkey, this.privkey);
                const resp = await request.get(url, { auth: { bearer: auth }, json: true, resolveWithFullResponse: true });
                if(resp.statusCode !== 200) reject(new Error(`Non-200 status code ${resp.statusCode}`));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body.data);
            } catch (e) {
                reject(e);
            }
        });
    }

    /**
     * Get a single user
     * @async
     * @param {String} id - ID of the user
     * @returns {PTUser?} - User
     * @throws {Error} - Any errors that happen
     */
    getUser(id) {
        return new Promise(async (resolve, reject) => {
            if(!id) reject(new Error('You mast pass a user ID!'));
            const url = `${this.url}/api/admin/users/${id}`;
            try {
                const auth = await this.getAuthHeader(url, '', this.pubkey, this.privkey);
                const resp = await request.get(url, { auth: { bearer: auth }, json: true, resolveWithFullResponse: true });
                if(resp.statusCode === 404) resolve(null);
                if(resp.statusCode !== 200) reject(new Error(`Non-200 status code ${resp.statusCode}`));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body.data);
            } catch (e) {
                reject(e);
            }
        });
    }


    /**
     * Creates a user
     * @async
     * @param {Object} user - User to add
     * @param {String} user.email - Email for new user
     * @param {String} user.username - Username for new user
     * @param {String} user.name_first - First name for new user
     * @param {String} user.name_last - Last name for new user
     * @param {String} user.password - Password for the new user
     * @param {Boolean} user.root_admin - Admin status for the new user
     * @returns {String} New user's ID
     */
    createUser(user) {
        return new Promise(async (resolve, reject) => {
            if(!user || !user.email || !user.username || !user.name_first || !user.name_last || !user.root_admin) reject(new Error('You must specify the user\'s email, username, first and last name, and admin status. If left blank, password will be generated by the server'));
            const url = this.url + '/users';
            try {
                const resp = await request.post(url, { body: user, json: true, resolveWithFullResponse: true });
                if(resp.statusCode === 422) reject(new Error(`Validation error on object ${user}`));
                if(resp.statusCode === 503) reject(new Error('Panel encountered an error while processing the request'));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body.id);
            } catch (e) {
                reject(e);
            }
        });
    }

    /**
     * Update a user
     * @async
     * @param {String} - ID of user to update
     * @param {Object} user - New user object
     * @param {String} user.username - User's new username
     * @param {String} user.email - User's new email
     * @param {String} user.name_first - User's new first name
     * @param {String} user.name_last - User's new last name
     * @param {String} user.password - User's new password
     * @param {String} user.root_admin - User's admin status
     * @returns {PTUser} The updated user
     * @throws {Error} Any errors
     */
    updateUser(id, user) {
        return new Promise(async (resolve, reject) => {
            const url = `${this.url}/admin/users/${id}`;
            try {
                const auth = await this.getAuthHeader(url, JSON.stringify(user), this.pubkey, this.privkey);
                const resp = await request.put(url, { auth: { bearer: auth }, json: true, resolveWithFullResponse: true, formData: user });
                if(resp.statusCode === 404) reject(new Error('Can\'t update non-existant user!'));
                if(resp.statusCode !== 200) reject(new Error(`Non-200 status code ${resp.statusCode}`));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body);
            } catch (e) {
                reject(e);
            }
        });
    }

    /**
     * Delete a user
     * @async
     * @param {String} id ID of user to delete
     * @throws {Error} Any errors that occur
     */
    deleteUser(id) {
        return new Promise(async (resolve, reject) => {
            if(!id) reject(new Error('Must specifiy a user ID'));
            const url = `${this.url}/admin/users/${id}`;
            try {
                const auth = await this.getAuthHeader(url, '', this.pubkey, this.privkey);
                const resp = await request.delete(url, { auth: { bearer: auth }, resolveWithFullResponse: true });
                if(resp.statusCode !== 204) reject(new Error(`Got a non-204 response: ${resp.statusCode}`));
                resolve();
            } catch (e) {
                reject(e);
            }
        });
    }

    // Server related functions

    /**
     * Get a list of servers
     * @async
     * @returns {PTServer[]} - Array of servers
     * @throws {Error} - Any errors that occur
     */
    getServers() {
        return new Promise(async (resolve, reject) => {
            const url = this.url + '/admin/servers';

                if(resp.statusCode !== 200) reject(new Error(`Got non-200 response: ${resp.statusCode}`));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body.data);
            } catch (e) {
                reject(e);
            }
        });
    }

    /**
     * Get a server
     * @async
     * @param {String} id - ID of server to get
     * @returns {PTServer} - Server
     * @throws {Error} - Any errors that occur
     */
    getServer(id) {
        return new Promise(async (resolve, reject) => {
            if(!id) reject(new Error('Missing id parameter'));
            const url = `${this.url}/admin/servers/${id}`;
            try {
                const auth = await this.getAuthHeader(url, '', this.pubkey, this.privkey);
                const resp = await request.get(url, { auth: { bearer: auth }, resolveWithFullResponse: true, json: true });
                if(resp.statusCode !== 200) reject(new Error(`Got non-200 response: ${resp.statusCode}`));
                if(resp.body.error) reject(new Error(`Error: ${resp.body.error}`));
                resolve(resp.body.data);
            } catch (e) {
                reject(e);
            }
        });
    }

    /**
     * Create a server
     * @async
     * @param {Object} server - Server to create
     * @param {String} server.name - Human-readable name for the server
     */
    createServer(server) {
        return new Promise(async (resolve, reject) => {
            if(!server) reject(new Error('You must specify all values for the new server'));
            const url = this.url + '/servers';
        });
    }

}
